  'SENTRY_LIST_ISSUES',
  project='api-prod',
  status='unresolved'
)

for issue in issues:
  issue['trace'] = run_composio_tool(
    'SENTRY_GET_EVENT',
    issue_id=issue['id']
  )

  f'Classify these {len(issues)} errors.'

)

  run_composio_tool(
    'LINEAR_CREATE_ISSUE',
    title=error['title'],
    team_id='ENG',
    priority=1
  )
